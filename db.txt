-----Users-----


create table users
(
    id int auto_increment primary key,
    username varchar(50)  not null,
    password varchar(255) not null,
    constraint username
        unique (username)
);



---------------


----Doctor------

CREATE TABLE doctor (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    phone_number VARCHAR(20),
    background TEXT
);
